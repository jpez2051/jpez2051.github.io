updateStreak() { const today = new Date(); const tKey = this.todayKey(); if (!this.state.lastDate) { this.state.streak = 1; } else { const lastDate = new Date(this.state.lastDate); const diffInDays = Math.floor((today - lastDate) / (1000 * 60 * 60 * 24)); if (diffInDays === 1) { this.state.streak += 1; // Successive day } else if (diffInDays > 1) { this.state.streak = 1; // Streak broken } // if diff is 0, do nothing (already added today) } this.state.lastDate = tKey; }
